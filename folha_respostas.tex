\NewDocumentCommand{\markerboxcorner}{O{5pt} m O{0.5pt} O{0.15} O{1}}{%
  \tikz[remember picture, overlay]{
    % Deslocamento horizontal
    \pgfmathsetlengthmacro{\dx}{#1}
    % Metade do comprimento da cruz
    \pgfmathsetlengthmacro{\halfsize}{#2/2}
    % Tamanho da caixa preta (meio lado)
    \pgfmathsetlengthmacro{\boxhalf}{#4}

    % Escopo deslocado horizontalmente
    \begin{scope}[xshift=\dx]
      % Desenha o quadrado conforme o formato
      \ifcase#5
        % zero não usado, para alinhar os casos (começa em 0)
      \or % case 1: quadrado preenchido (default)
        \fill[black] (-\boxhalf,-\boxhalf) rectangle (\boxhalf,\boxhalf);
      \or % case 2: quadrado vazado (contorno)
        \draw[line width=#3] (-\boxhalf,-\boxhalf) rectangle (\boxhalf,\boxhalf);
      \or % case 3: quadrado com círculo (contorno) no centro
        \draw[line width=#3] (-\boxhalf,-\boxhalf) rectangle (\boxhalf,\boxhalf);
        \draw[line width=#3] (0,0) circle [radius=0.6*\boxhalf];
      \or % case 4: quadrado com dot (círculo preenchido) no centro
        \draw[line width=#3] (-\boxhalf,-\boxhalf) rectangle (\boxhalf,\boxhalf);
        \fill (0,0) circle [radius=0.4*\boxhalf];
      \or % case 5: quadrado com círculo (contorno) e dot preenchido no centro
        \draw[line width=#3] (-\boxhalf,-\boxhalf) rectangle (\boxhalf,\boxhalf);
        \draw[line width=#3] (0,0) circle [radius=0.76*\boxhalf];
        \fill (0,0) circle [radius=0.35*\boxhalf];
      \fi

      % Cruz centrada na caixa (sempre desenhada)
      \draw[line width=#3]
        (0, -\halfsize) -- (0, \halfsize);   % vertical
      \draw[line width=#3]
        (-\halfsize, 0) -- (\halfsize, 0);   % horizontal
    \end{scope}
  }
}
